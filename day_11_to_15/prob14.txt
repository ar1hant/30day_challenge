class Solution{
public:
    int minSteps(int D){
        int steps = 1;
        int sum = 0;
        while(sum < D)
            sum += steps++;
        while((sum - D) % 2)
            sum += steps++;
        return steps - 1;
    }
};

// { Driver Code Starts.

int main(){
    int t;
    cin>>t;
    while(t--){
        int D;
        cin>>D;
        
        Solution ob;
        cout<<ob.minSteps(D)<<"\n";
    }
    return 0;
}  // } Driver Code Ends
