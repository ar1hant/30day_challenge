#include <bits/stdc++.h>

using namespace std;

class Solution {
  public:
    int findPosition(int N , int M , int K) {
        int temp = N - K + 1;
        if(M <= temp)
            return M + K - 1;
        M = (M - temp) % N;
        return M ? M : N;
    }
};

// Time compleity of this program O(1)
// Extra Space compleity of this program O(1)

int main() {
    int t;
    cin >> t;
    while (t--) {
        int N,M,K;
        
        cin>>N>>M>>K;

        Solution ob;
        cout << ob.findPosition(N,M,K) << endl;
    }
    return 0;
}
