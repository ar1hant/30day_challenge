#include<iostream>

using namespace std;

class Solution {
	public:
	    int mod = 1e9 + 7;
		int CountWays(string str){
		    if(str[0] == '0')
		        return 0;
		    int n = str.length();
		    vector<int> dp(n + 1, 0);
		    dp[0] = 1;
		    dp[1] = 1;
		    for(int i = 2;i <= n;i++){
		        if(str[i - 1] > '0')
		            dp[i] = dp[i - 1];
		        if(str[i - 2] == '1' || (str[i - 2] == '2' && str[i - 1] < '7'))
		            dp[i] = (dp[i] + dp[i - 2]) % mod;
		    }
		    return dp[n];
		}
};

// Time compleity of this program O(n)
// Extra Space compleity of this program O(n)

int main(){
	int tc;
	cin >> tc;
	while(tc--){
		string str;
		cin >> str;
		Solution obj;
		int ans = obj.CountWays(str);
		cout << ans << "\n";
	}
	return 0;
}
