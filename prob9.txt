#include<bits/stdc++.h> 

using namespace std; 

class Solution{   
public:
    string printMinNumberForPattern(string s){
        int n = s.length();
        if(n == 1){
            if(s[0] == 'I')
                return "12";
            return "21";
        }
        
        string res = "1";
        for(int i = 0;i < n;i++)
            res.push_back('0' + i + 2);
            
        int i = 0, j = 1;
        while(j <= n){
            if(s[j - 1] == 'I'){
                reverse(res.begin() + i, res.begin() + j);
                i = j;
            }
            j++;
        }
        if(s.back() == 'D')
            reverse(res.begin() + i, res.begin() + j);
        
        return res;
    }
};

// Time compleity of this program O(n)
// Extra Space compleity of this program O(1)

int main() 
{ 
    int t;
    cin>>t;
    while(t--)
    {
        string S;
        cin >> S;
        Solution ob;
        cout << ob.printMinNumberForPattern(S) << endl;
    }
    return 0; 
}
