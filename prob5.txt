#include <bits/stdc++.h>

using namespace std;

#define ull unsigned long long


class Solution{
public:	
	ull getNthUglyNo(int n) {
	    set<long long> s;
	    s.insert(1);
	    while(--n){
	        auto it = s.begin();
	        long long temp = *it;
	        s.erase(it);
	        s.insert(temp * 2);
	        s.insert(temp * 3);
	        s.insert(temp * 5);
	    }
	    auto it = s.begin();
	    return *it;
	}
};

// Time compleity of this program O(n)
// Extra Space compleity of this program O(n)

int main() {
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        Solution ob;
        auto ans = ob.getNthUglyNo(n);
        cout << ans << "\n";
    }
    return 0;
}
