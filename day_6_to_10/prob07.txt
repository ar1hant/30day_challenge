#include<bits/stdc++.h>

using namespace std;

class Solution {
	public:
		string FirstNonRepeating(string s){
		    int n = s.length();
		    if(n == 1)
		        return s;
		    vector<int> v(26, 0);
		    string res;
		    queue<int> q;
		    for(int i = 0;i < n;i++){
		        int pos = s[i] - 'a';
		        v[pos]++;
		        if(v[pos] == 1)
		            q.push(pos);
		        while(!q.empty() && v[q.front()] != 1)
		            q.pop();
		        if(q.empty())
		            res.push_back('#');
		        else
		            res.push_back(q.front() + 'a');
		    }
		    return res;
		}
};

// Time compleity of this program O(n)
// Extra Space compleity of this program O(1)

int main(){
	int tc;
	cin >> tc;
	while(tc--){
		string A;
		cin >> A;
		Solution obj;
		string ans = obj.FirstNonRepeating(A);
		cout << ans << "\n";
	}
	return 0;
}
