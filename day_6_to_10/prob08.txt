#include<bits/stdc++.h>

using namespace std;

class Solution
{
    public:
    long long countWays(int n){
        return ((n / 2) + 1);
    }
};

//Time compleity of this program O(1)
//Extra Space compleity of this program O(1)

int main()
{
    int t;
    cin >> t;
    
    while(t--)
    {
        //taking stair count
        int m;
        cin>>m;
        Solution ob;
        cout<<ob.countWays(m)<<endl;
    }
    return 0;
}
