#include <bits/stdc++.h>

using namespace std;

int minSubArrayLen(int target, vector<int>& nums) {
    int res = INT_MAX;
    int temp = 0;
    int sum = 0;
    for(int i = 0;i < nums.size();i++){
      sum += nums[i];
      while(sum >= target){
        res = min(res, i + 1 - temp);
        sum -= nums[temp++];
      }
    }
    return (res == INT_MAX ) ? 0 : res;
}

int main(){
  int n, target, x;
  cin >> n >> target;
  vector<int> v;
  for(int i = 0;i < n;i++){
    cin >> x;
    v.push_back(x);
  }
  cout << minSubArrayLen(target, v);
}
